1. Post - Order


```
class Solution:
    def flatten(self, root: TreeNode) -> None:
        stack, temp = [], []
        cur = root
        while True:
            while cur:
                temp.append(cur)
                stack.append(cur)
                cur = cur.left
            if not stack:
                break
            cur = stack.pop()
            cur = cur.right
        for i in range(1, len(temp)):
            temp[i - 1].left = None
            temp[i - 1].right = temp[i]
        return temp[0] if temp else []
```
