### 1\. Using Counter

```
from typing import *
from collections import Counter

class Solution:
    def topKFrequent(self, nums: List[int], k: int) -> List[int]:
        counter_dict = Counter()
        for num in nums:
            counter_dict[num] += 1
        return list(map(lambda x: x[0], counter_dict.most_common(k)))[:k]
```

### 2\. Bucket Sort

```
from typing import *
from collections import defaultdict

class Solution:
    def topKFrequent(self, nums: List[int], k: int) -> List[int]:
        dict1 = defaultdict(int)
        for num in nums:
            dict1[num] += 1

        bucket = []
        for _ in range(len(nums) + 1):
            bucket.append([])
        for key in dict1:
            value = dict1[key]
            if not bucket[value]:
                bucket[value] = [key]
            else:
                bucket[value].append(key)

        result = []
        for i in range(len(nums), -1, -1):
            if bucket[i]:
                result += bucket[i]
                if len(result) >= k:
                    break
        return result
```
