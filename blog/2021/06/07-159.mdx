```
from typing import *

class Solution:
    def generate(self, numRows: int) -> List[List[int]]:
        answer = [[1]]
        if numRows == 1:
            return answer
        elif numRows == 2:
            answer.append([1, 1])
            return answer
        else:
            answer.append([1, 1])
            for index in range(2, numRows):
                cur, temp = 0, []
                while cur < index + 1:
                    if cur in [0, index]:
                        temp.append(1)
                    else:
                        temp.append(answer[index - 1][cur - 1] + answer[index - 1][cur])
                    cur += 1
                answer.append(temp)
            return answer
```
